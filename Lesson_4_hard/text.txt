//необходимо создать и заполнить массив вида arr[i,j]
//где i первый идентификатор интервала, z конец итервала,(i,j <= z) (j = i; j<z; j++) => (arr[1,2] arr[1,3] arr[1,4] ...arr[i,j]) 
//нужен цикл который пушит такие пары в массив в интервале от i до z
//

let arr = [];

for (let i = 1, z = 10; i <= z; i++) {
	for (j = i; j <= z; j++) {
		arr.push(j,i)
	}
}
console.log(arr);
